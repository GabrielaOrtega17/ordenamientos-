#ifndef ORDENAMIENTO_H_INCLUDED
#define ORDENAMIENTO_H_INCLUDED

#include <iostream>

using namespace std;

const int TAM=100;

class Estudiante {
private:
    std::string nombre;
    int materias;
    std::string carrera;
    float promedio;
public:
    Estudiante();
    Estudiante(const Estudiante&);
    Estudiante(std::string n, int e);
    Estudiante& operator=(Estudiante&);
    bool operator == (const Estudiante&) const;
    bool operator != (const Estudiante&) const;
    bool operator < (const Estudiante&) const;
    bool operator <= (const Estudiante&) const;
    bool operator > (const Estudiante&) const;
    bool operator >= (const Estudiante&) const;
    friend std::ostream& operator<<( std::ostream&, Estudiante&);
    friend std::istream& operator>>( std::istream&, Estudiante&);
};


class Lista{
private:
    int tam;
    int cont;
    int ult;
    Estudiante datos[100];

public:
    Lista(){};
    bool posValida (const int&);
    void enqueue(Estudiante& elem);
    int inserta(Estudiante&, int&);
    void elimina(const int&);
    void agrega(Estudiante&);
    int busca(const Estudiante&);
    void muestra();
    bool vacia()const;
    bool llena()const;
    int buscaBinaria(const Estudiante&);
    int ultimo()const;
    friend std::ostream& operator<<(std::ostream & o, Lista& L);
    void ordenarInser();
};


#endif // ORDENAMIENTO_H_INCLUDED
