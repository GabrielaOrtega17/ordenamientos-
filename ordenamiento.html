    //posicion
    void Lista::elimina(const int& p){
        if(!posValida(p)){
            cout << " Posicion invalida..." << endl << endl;
        }
        int x(p);
        while(x < cont){
            datos[x] = datos[x + 1];
            x++;
        }
        cont--;
    }
    void Lista::agrega(Estudiante& e){
        if(llena()){
            cout << " La lista esta llena ";
        }
        else{
            datos[cont] = e;
            cont++;
        }
    }

    int Lista::busca(const Estudiante& e){
        if(vacia()){
            return -1;
        }
       for(int i=0; i<=ult; i++){
        if(e==datos[i]){
            return i;
        }
        return -1;
       }
    }

        if(vacia()){

    int Lista::buscaBinaria(const Estudiante& e){
    int x(0), y(ult), m;

    while(x <= y){
        m = (x + y) / 2;
        if(datos[m] == e){
            return m;
        }
        if(e < datos[m]){
            y = m - 1;
        }
        else{
            x = m + 1;
        }
    }
    return -1;
}

}

void Lista::ordenarInser(){
    int i(1), j;
    Estudiante aux;

    while(i <= ult){
        aux = datos[i];
        j = i;

        while(j < 0 && aux < datos[j - 1]){
            datos[j] = datos[j - 1];
            j--;
        }
        if(i != j){
            datos[j] = aux;
        }

        i++;
    }
}
